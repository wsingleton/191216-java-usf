CREATE TABLE useracct(
    user_id            NUMBER(5),   
    acct_id           NUMBER(5),  
    
    CONSTRAINT   pk_useracct   
    PRIMARY KEY(user_id),
    
    CONSTRAINT    fk_useracct   
    FOREIGN KEY(acct_id)
    REFERENCES accounts(acct_id)
);

CREATE TABLE accounts(
    acct_id           NUMBER(5)   CONSTRAINT   pk_accounts   PRIMARY KEY,
    balance           NUMBER(10, 2)   
);

CREATE TABLE users(
    user_id          NUMBER(5)  CONSTRAINT pk_users PRIMARY KEY,
    user_fn          VARCHAR2(25),
    user_ln          VARCHAR2(25),
    user_name        VARCHAR2(25),
    user_pass        VARCHAR2(25)
);
